//@Repository
public class SeikyusienAPIQueueRepositoryImpl implements SeikyusienAPIQueueRepository {

    /**
     * queueconfpath
     */
    @Value("${QUEUE_CONF_PATH:default}")
    private String queueconfpath;

    /**
     * リトライ時機処理
     * @param waitTime
     */
    private void sleepWithInterruptHandling(int waitTime) {
        try {
            Thread.sleep(waitTime);
        } catch (InterruptedException ie) {
            Thread.currentThread().interrupt();
            throw new QueueException("スレッドが中断されました", ie);
        }
    }

    /** ログ */
    private static final Logger LOGGER = LoggerFactory.getLogger(SeikyusienAPIQueueRepositoryImpl.class);

    /**
     * sqs設定ファイル情報読込用
     */
    private final XmlReader xmlReader;

    public SeikyusienAPIQueueRepositoryImpl(XmlReader xmlReader) {
        this.xmlReader = xmlReader;
    }

    /** メッセージ開始タグ */
    private static final String MESSAGE_START_TAG = "<Message>";
    /** メッセージ終了タグ */
    private static final String MESSAGE_END_TAG = "</Message>";

    /**
     * {@inheritDoc}
     */
    @Override
    public boolean send(String queueKey, String program, String programArgs, String reqSvr,
            List<RCqsndExecEnvInfo> envInfoList) {

        // 排他キー空欄でキュー送信要求を呼び出し
        return send(queueKey, program, programArgs, reqSvr, envInfoList, "");
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public boolean send(String queueKey, String program, String programArgs, String reqSvr,
            List<RCqsndExecEnvInfo> envInfoList, String groupId) {
        LOGGER.debug("START");

        boolean result = false;
        try {
            result = sendQueue(queueKey, program, programArgs, reqSvr, envInfoList, groupId);
        } catch (Exception e) {
            String logmsg = String.format("キュー登録に失敗しました。 %s", e.getMessage());
            LOGGER.error(logmsg, e);

            // 引数．環境切替情報を半角スペース区切りの文字列への変換を実施する。

            StringBuilder envSwitchInfoBuilder = new StringBuilder();
            for (RCqsndExecEnvInfo envInfo : envInfoList) {
                envSwitchInfoBuilder.append(envInfo.getEnvName()).append("");
                envSwitchInfoBuilder.append(envInfo.getEnvValue()).append("");
            }
            String envSwitchInfo = "";
            if (envSwitchInfoBuilder.length() != 0) {
                envSwitchInfo = envSwitchInfoBuilder.substring(0, envSwitchInfoBuilder.length() - 1);
            }

            String from = "\\$";
            String to = "$";
            envSwitchInfo = envSwitchInfo.replace(from, to);

            // リランコマンド編集
            // "python3 $AP_HOME/bin/apInfra/py/apExecSvrReqQueue.py '引数.キューノード'
            // '引数.起動プログラム' '引数.起動プログラムに渡す引数' '引数.要求先' '引数.環境情報を半角スペース区切りに変換した文字列' '引数.排他キー'
            from = "'";
            to = "'\"'\"'";
            programArgs = programArgs.replace(from, to);
            String rerunCmd = "python3 $AP_HOME/bin/apInfra/py/apExecSvrReqQueue.py" + " '" + queueKey + "' '" + program
                    + "' '" + programArgs + "' '" + reqSvr + "' '" + envSwitchInfo + "' '" + groupId + "'";

            // 業務ログにエラーログを出力する
            logmsg = String.format("リランコマンド：%s", rerunCmd);
            LOGGER.warn(logmsg);
            result = false;
        }
        LOGGER.debug("END");

        // 処理結果に"0"(正常終了)を設定し、返却する。
        return result;
    }

    /**
     * キュー送信要求(内部)
     *
     * @param queueKey    キューノード
     * @param program     起動プログラム
     * @param programArgs 起動プログラムに渡す引数
     * @param reqSvr      要求先
     * @param envInfoList 環境切替情報
     * @param groupId     排他キー
     * @return 処理結果（0:正常終了 / -1:システムエラー）
     * @throws UnsupportedEncodingException
     * @throws NoSuchAlgorithmException
     * @throws KeyManagementException
     */

    private boolean sendQueue(String queueKey, String program, String programArgs, String reqSvr,
            List<RCqsndExecEnvInfo> envInfoList, String groupId)
            throws NoSuchAlgorithmException, KeyManagementException, IOException {

        // 接続先URL取得
        String url = getQueueUrl(queueKey);
        // リトライ設定取得
        int[] retryConfig = getRetryConfig();
        // リクエストボディ生成
        String requestBody = buildRequestBody(program, programArgs, reqSvr, envInfoList, groupId);
        // SSL設定
        SSLSocketFactory sslFactory = createSslFactory();
        // HTTP通信実行
        executeWithRetry(url, requestBody, retryConfig, sslFactory);
        return true;
    }

    /**
     * 接続先URL取得
     *
     * @param queueKey キューノード
     * @return 接続先URL
     */
    private String getQueueUrl(String queueKey) {
        // 設定ファイルから接続先URLを取得
        String url = xmlReader.getBizUrl(queueKey);
        // URL取得に失敗した場合
        if (url == null || url.isEmpty()) {
            String logmsg = String.format("設定取得に失敗しました。引数として渡されたキー:%s", queueKey);
            throw new QueueException(logmsg);
        }
        return url;
    }

    /**
     * リトライ設定取得
     *
     * @return リトライ設定情報配列 [リトライ回数, リトライ待機時間, タイムアウト時間]
     */
    private int[] getRetryConfig() {
        // リトライ回数初期値
        int retryCountValue = 3;
        // リトライ待機時間初期値
        int retryWaitTimeValue = 1000;
        // タイムアウト時間初期値
        int timeout = 30000;

        // リトライ回数取得
        String propertyValue = xmlReader.getretryInfo("retrySqsCount");
        if (StringUtils.isNumeric(propertyValue)) {
            retryCountValue = Integer.parseInt(propertyValue);
        }
        // リトライ待機時間取得
        propertyValue = xmlReader.getretryInfo("retrySqsWaitTime");
        if (StringUtils.isNumeric(propertyValue)) {
            retryWaitTimeValue = Integer.parseInt(propertyValue) * 1000;
        }
        // タイムアウト時間取得
        propertyValue = xmlReader.getretryInfo("retrySqsTimeOut");
        if (StringUtils.isNumeric(propertyValue)) {
            timeout = Integer.parseInt(propertyValue) * 1000;
        }
        return new int[] { retryCountValue, retryWaitTimeValue, timeout };
    }

    /**
     * リクエストボディ生成
     *
     * @param program     起動プログラム
     * @param programArgs 起動プログラムに渡す引数
     * @param reqSvr      要求先
     * @param envInfoList 環境切替情報
     * @param groupId     排他キー
     * @return リクエストボディ
     * @throws UnsupportedEncodingException
     */
    private String buildRequestBody(String program, String programArgs, String reqSvr,
            List<RCqsndExecEnvInfo> envInfoList, String groupId) throws UnsupportedEncodingException {

        // 引数．環境切替情報を半角スペース区切りの文字列へ変換
        String envSwitchInfo = buildEnvSwitchInfo(envInfoList);

        // 起動プログラムに渡す引数の"（ダブルクォーテーション）についてエスケープ処理（\"）を実施
        String programArgs4Json = programArgs.replace("\"", "\\\"");

        // UUID形式の一意の値を生成
        String uniqueId = UUID.randomUUID().toString();

        // 引数.排他キーが空の場合、排他キーに生成した一意の値を設定する
        String exKey = (groupId == null || groupId.isEmpty()) ? uniqueId : groupId;

        // メッセージボディを生成する
        String messageBody = new StringBuilder().append("{\"program\": \"").append(program)
                .append("\", \"program_arg\": \"").append(programArgs4Json).append("\", \"req_svr\": \"").append(reqSvr)
                .append("\", \"env_info_list\": \"").append(envSwitchInfo).append("\", \"group_id\": \"")
                .append(groupId).append("\"}").toString();

        // メッセージボディにURLエンコード処理を実施する
        String encodeBody = URLEncoder.encode(messageBody, "UTF-8");

        // リクエストボディを生成する
        return new StringBuilder().append("Action=SendMessage&MessageBody=").append(encodeBody)
                .append("&MessageGroupId=").append(exKey).append("&MessageDeduplicationId=").append(uniqueId)
                .append("&Version=2012-11-05").toString();
    }

    /**
     * 環境切替情報文字列生成
     *
     * @param envInfoList 環境切替情報
     * @return 環境切替情報文字列
     */
    private String buildEnvSwitchInfo(List<RCqsndExecEnvInfo> envInfoList) {
        // 引数．環境切替情報を半角スペース区切りの文字列への変換を実施する
        StringBuilder sb = new StringBuilder();
        for (RCqsndExecEnvInfo envInfo : envInfoList) {
            sb.append(envInfo.getEnvName()).append(" ");
            sb.append(envInfo.getEnvValue()).append(" ");
        }
        String result = sb.length() != 0 ? sb.substring(0, sb.length() - 1) : "";
        // 環境変数情報の"\$"を"$"に置換
        return result.replace("\\$", "$");
    }

    /**
     * SSL設定
     *
     * @return SSLSocketFactory
     * @throws NoSuchAlgorithmException
     * @throws KeyManagementException
     */
    private SSLSocketFactory createSslFactory() throws NoSuchAlgorithmException, KeyManagementException {
        // SSL設定
        SSLContext sslCtx = SSLContext.getInstance("TLS");
        TrustManager[] nonAuthMng = { new X509TrustManager() {
            @Override
            public void checkClientTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
                // 開発環境のため、クライアント証明書の検証を行わない
            }

            @Override
            public void checkServerTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
                // 開発環境のため、サーバー証明書の検証を行わない
            }

            @Override
            public X509Certificate[] getAcceptedIssuers() {
                return new X509Certificate[0];
            }
        } };
        sslCtx.init(null, nonAuthMng, null);
        return sslCtx.getSocketFactory();
    }

    /**
     * HTTP通信実行（リトライあり）
     *
     * @param url         接続先URL
     * @param requestBody リクエストボディ
     * @param retryConfig リトライ設定情報
     * @param sslFactory  SSLSocketFactory
     * @throws IOException
     */
    private void executeWithRetry(String url, String requestBody, int[] retryConfig, SSLSocketFactory sslFactory)
            throws IOException {

        // リトライ設定取得
        int retryCountValue = retryConfig[0];
        int retryWaitTimeValue = retryConfig[1];
        int timeout = retryConfig[2];

        // リトライ回数初期化
        int retryNow = 0;

        while (true) {
            HttpsURLConnection con = null;
            try {
                // 設定ファイルから取得した接続先URLへHTTPリクエストを送信し、処理結果をHTTPレスポンスで受信する
                con = openConnection(url, timeout, sslFactory);
                // ボディ部設定
                sendRequest(con, requestBody);
                // HTTPステータスコード取得
                int httpStatusCode = con.getResponseCode();

                // HTTPステータスコードが200の場合
                if (httpStatusCode == 200) {
                    break;
                }
                // HTTPステータスコードが200以外
                retryNow = handleResponse(httpStatusCode, retryNow, retryCountValue, retryWaitTimeValue, con);

            } catch (UnknownHostException | SocketTimeoutException | ConnectException e) {
                // 通信エラーの場合
                retryNow = handleNetworkError(e, retryNow, retryCountValue, retryWaitTimeValue);
            } finally {
                // 接続を切断
                if (con != null) {
                    con.disconnect();
                }
            }
        }
    }

    /**
     * HTTP接続処理
     *
     * @param url        接続先URL
     * @param timeout    タイムアウト時間
     * @param sslFactory SSLSocketFactory
     * @return HttpsURLConnection
     * @throws IOException
     */
    private HttpsURLConnection openConnection(String url, int timeout, SSLSocketFactory sslFactory) throws IOException {

        HttpsURLConnection.setDefaultHostnameVerifier((hostname, session) -> {
            if ("localhost".equals(hostname)) {
                // 開発環境用のため、ホスト名検証を行わない
                return true;
            }
            return HttpsURLConnection.getDefaultHostnameVerifier().verify(hostname, session);
        });

        // 接続先URLへHTTPリクエストを送信
        HttpsURLConnection con = (HttpsURLConnection) URI.create(url).toURL().openConnection();
        con.setRequestMethod("POST");
        con.setRequestProperty("Content-Type", "application/x-www-form-urlencoded; charset=shift_jis");
        con.setConnectTimeout(timeout);
        con.setReadTimeout(timeout);
        con.setDoOutput(true);
        con.setSSLSocketFactory(sslFactory);
        con.connect();
        return con;
    }

    /**
     * リクエスト送信処理
     *
     * @param con         HttpsURLConnection
     * @param requestBody リクエストボディ
     * @throws IOException
     */
    private void sendRequest(HttpsURLConnection con, String requestBody) throws IOException {
        // ボディ部設定
        try (PrintStream ps = new PrintStream(con.getOutputStream())) {
            ps.print(requestBody);
        }
    }

    /**
     * HTTPレスポンス処理
     *
     * @param httpStatusCode     HTTPステータスコード
     * @param retryNow           現在のリトライ回数
     * @param retryCountValue    最大リトライ回数
     * @param retryWaitTimeValue リトライ待機時間
     * @param con                HttpsURLConnection
     * @return リトライ回数
     * @throws IOException
     */
    private int handleResponse(int httpStatusCode, int retryNow, int retryCountValue, int retryWaitTimeValue,
            HttpsURLConnection con) throws IOException {

        // 通信異常、HTTPステータスコードが403,429,500,502,503,504,509
        // かつリトライ回数が設定ファイルから取得した回数以下
        if (retryNow < retryCountValue && isRetryableStatus(httpStatusCode)) {
            // リトライ回数加算
            retryNow++;
            // 業務ログにワーニングログを出力する
            String logmsg = String.format("キュー登録に失敗しました。リトライします。HTTPステータスコード:%s", httpStatusCode);
            LOGGER.warn(logmsg);
            // 設定したリトライ待機時間だけ処理を待つ
            sleepWithInterruptHandling(retryWaitTimeValue);
            return retryNow;
        }
        // 業務ログにエラーログを出力する
        String returnMessage = readErrorMessage(con);
        String logmsg = String.format("HTTPステータスコード:%s エラー内容:%s", httpStatusCode, returnMessage);
        throw new QueueException(logmsg);
    }

    /**
     * リトライ可能なHTTPステータスコードかチェック
     *
     * @param httpStatusCode HTTPステータスコード
     * @return true:リトライ可能 / false:リトライ不可
     */
    private boolean isRetryableStatus(int httpStatusCode) {
        // リトライ対象のHTTPステータスコード: 403,429,500,502,503,504,509
        return httpStatusCode == 429 || httpStatusCode == 403 || httpStatusCode == 500 || httpStatusCode == 502
                || httpStatusCode == 503 || httpStatusCode == 504 || httpStatusCode == 509;
    }

    /**
     * エラーメッセージ取得
     *
     * @param con HttpsURLConnection
     * @return エラーメッセージ
     * @throws IOException
     */
    private String readErrorMessage(HttpsURLConnection con) throws IOException {
        // エラーストリームを取得
        InputStream is = con.getErrorStream();
        if (is == null) {
            // エラーストリームを取得できない場合、インプットストリームを取得
            is = con.getInputStream();
        }
        // 取得したストリームを読込、処理結果を取得
        StringBuilder sb = new StringBuilder();
        try (BufferedReader br = new BufferedReader(new InputStreamReader(is))) {
            String s;
            while ((s = br.readLine()) != null) {
                sb.append(s);
            }
        }
        // 処理結果内から<Message>～</Message>に囲まれた値を取得
        String returnMessage = sb.toString();
        int messageStartPoint = returnMessage.indexOf(MESSAGE_START_TAG);
        int messageEndPoint = returnMessage.indexOf(MESSAGE_END_TAG);
        if (messageStartPoint != -1 && messageEndPoint != -1) {
            returnMessage = returnMessage.substring(messageStartPoint + MESSAGE_START_TAG.length(), messageEndPoint);
        }
        return returnMessage;
    }

    /**
     * ネットワークエラー処理
     *
     * @param e                  例外
     * @param retryNow           現在のリトライ回数
     * @param retryCountValue    最大リトライ回数
     * @param retryWaitTimeValue リトライ待機時間
     * @return リトライ回数
     */
    private int handleNetworkError(Exception e, int retryNow, int retryCountValue, int retryWaitTimeValue) {

        if (retryNow < retryCountValue) {
            // リトライ回数加算
            retryNow++;
            // 業務ログにワーニングログを出力する
            String logmsg = String.format("キュー登録に失敗しました。リトライします。HTTPステータスコード:- エラー内容:%s", e.getMessage());
            LOGGER.warn(logmsg);
            // 設定したリトライ待機時間だけ処理を待つ
            sleepWithInterruptHandling(retryWaitTimeValue);
            return retryNow;
        }
        // 業務ログにエラーログを出力する
        String logmsg = String.format("HTTPステータスコード:- エラー内容:%s", e.getMessage());
        throw new QueueException(logmsg, e);
    }
