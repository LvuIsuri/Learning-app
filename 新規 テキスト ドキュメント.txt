import java.util.zip.ZipOutputStream
import java.util.zip.ZipEntry

def pref = vars.get("pref")
def kikan = vars.get("kikan")
def seiqiraino = "5071" + pref + kikan + "1001"
def filename = "202612_1" + seiqiraino + ".zip"
def filePath = "C:/test/tmp/" + filename

new File("C:/test/tmp").mkdirs()

def fos = new FileOutputStream(filePath)
def zos = new ZipOutputStream(fos)
zos.putNextEntry(new ZipEntry("dummy.txt"))
zos.write("test data pref=${pref} kikan=${kikan}".getBytes())
zos.closeEntry()
zos.close()
fos.close()

vars.put("seiqiraino", seiqiraino)
vars.put("filename", filename)
vars.put("filePath", filePath)

log.info("Generated: " + filePath)
